// USAGE EXAMPLE

<!-- Include from CDN -->
<script src="https://unpkg.com/cybercore-charts"></script>

<script>
  // Create a chart using the factory function
  const chart = CyberCharts.createChart('line', '#my-chart', {
    data: [
      { x: 'Jan', y: 10 },
      { x: 'Feb', y: 25 },
      { x: 'Mar', y: 15 },
      { x: 'Apr', y: 30 },
    ],
    theme: 'cyan',
    showArea: true,
    glow: true,
  });

  // Update data later
  chart.update(newData);

  // Listen to events
  chart.on('pointClick', (event) => {
    console.log('Clicked:', event.data);
  });

  // Clean up when done
  chart.destroy();
</script>
